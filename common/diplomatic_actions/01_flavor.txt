make_dominion = {
	requires_approval = no
	should_notify_third_parties = no
	show_effect_in_tooltip = no

	potential = {
		NOT = { is_country_type = decentralized }
		scope:target_country = {
			NOT = { is_country_type = decentralized }
		}
		is_player = yes
	}
	
	possible = {
		is_at_war = no
		is_diplomatic_play_committed_participant = no
		is_revolutionary = no
		scope:target_country = {
			is_subject_of = ROOT
			is_at_war = no
			is_diplomatic_play_committed_participant = no
			is_revolutionary = no

			has_diplomatic_pact = {
				who = ROOT
				type = puppet
			}
		}
	}
	
	accept_effect = {
		scope:target_country = {
		change_subject_type = subject_type_dominion
		}
	}
	
	ai = {
		will_propose = { 
			always = no
		}
	}
}

invest_fdi = {
	requires_approval = no
	should_notify_third_parties = no
	show_effect_in_tooltip = no

	potential = {
		NOT = { is_country_type = decentralized }
		scope:target_country = {
			NOT = { is_country_type = decentralized }
		}
		OR = {
			is_player = yes
			AND = {
			country_rank > rank_value:major_power 
			"relations(scope:target_country)" >= 20
			weekly_net_fixed_income > 1000
			}
			AND = {
			scope:target_country = {
				is_subject_of = ROOT
			}
			weekly_net_fixed_income > 1000
			}
		}
	}
	
	possible = {
		is_at_war = no
		is_diplomatic_play_committed_participant = no
		is_revolutionary = no
		scope:target_country = {
			is_at_war = no
			is_diplomatic_play_committed_participant = no
			is_revolutionary = no
			country_rank <= rank_value:major_power 

			NOT = {
				has_variable = taker
			}
		}
		gold_reserves > 50000
		has_interest_marker_in_region = scope:target_country.capital.region
	}
	
	accept_effect = {
		add_treasury = -50000
		scope:target_country = {
			add_investment_pool = 50000
			save_scope_as = taker 
			set_variable = taker
			add_modifier = fdi_modifier
		}
		change_relations = {
		country = scope:target_country
		value = 10
		}

		save_scope_as = giver
		set_variable = giver

		trigger_event = { id = fdi.1 
		years = 1}
	}
	
	ai = {
		will_propose = { 
			always = yes
		}
	}
}

occupy_split = {
	requires_approval = no
	should_notify_third_parties = yes
	show_effect_in_tooltip = yes

	potential = {
		is_at_war = no
		is_diplomatic_play_committed_participant = no
		is_revolutionary = no
		scope:target_country = {
			is_revolutionary = no
		}
		NOT = { is_country_type = decentralized }
		scope:target_country = {
			NOT = { is_country_type = decentralized }
		}
	}
	
	possible = {
		is_at_war = no
		is_diplomatic_play_committed_participant = no
		is_revolutionary = no
		scope:target_country = {
			is_revolutionary = no
		}

		NOT = {
			has_variable = occupiedcountry
		}
		NOT = { is_country_type = decentralized }
		scope:target_country = {
			NOT = { is_country_type = decentralized }
		}

		scope:target_country = {
			any_scope_state = {
            OR = {
                    is_split_state = yes 
                    is_treaty_port = yes
            }
            state_region = {
                any_scope_state = { owner = ROOT }
            }

            NOT = { owner = ROOT }
            owner = {
                    is_revolutionary = no
                }
        }
		}
	}
	
	accept_effect = {
        change_infamy = 10

		set_variable = { name = occupiedcountry days = 365 } 

		save_scope_as = occupier

		scope:target_country = {
			random_scope_state = {
				limit = {
					OR = {
						is_split_state = yes 
						is_treaty_port = yes
           			}
					state_region = {
						any_scope_state = { owner = ROOT }
					}

					NOT = { owner = ROOT }
				}

				state_region = {
                	add_claim = scope:target_country
                	add_claim = scope:target_country.top_overlord

					save_scope_as = captured_state_region
            	}

				scope:target_country.top_overlord = {
					save_scope_as = enemy
				}

            	set_state_owner = ROOT
				save_scope_as = to_be_captured

				scope:target_country.top_overlord = {
						trigger_event = { id = fdi.3
						days = 1}
					}
				}
			}

		change_relations = {
		country = scope:target_country
		value = -100
		}

		

		
	}
	
	ai = {
		will_propose = { 
			always = no
		}
	}
}

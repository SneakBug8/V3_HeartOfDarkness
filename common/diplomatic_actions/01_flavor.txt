

invest_fdi = {
	requires_approval = no
	should_notify_third_parties = no
	show_effect_in_tooltip = no

	potential = {
		NOT = { is_country_type = decentralized }
		scope:target_country = {
			NOT = { is_country_type = decentralized }
		}
		OR = {
			is_player = yes
			AND = {
			country_rank > rank_value:major_power 
			"relations(scope:target_country)" >= 20
			weekly_net_fixed_income > 1000
			}
			AND = {
			scope:target_country = {
				is_subject_of = ROOT
			}
			weekly_net_fixed_income > 1000
			}
		}
	}
	
	possible = {
		is_at_war = no
		is_diplomatic_play_committed_participant = no
		is_revolutionary = no
		scope:target_country = {
			is_at_war = no
			is_diplomatic_play_committed_participant = no
			is_revolutionary = no
			country_rank <= rank_value:major_power 

			NOT = {
				has_variable = taker
			}
		}
		gold_reserves > 50000
		has_interest_marker_in_region = scope:target_country.capital.region
	}
	
	accept_effect = {
		add_treasury = -50000
		scope:target_country = {
			add_investment_pool = 50000
			save_scope_as = taker 
			set_variable = taker
			add_modifier = fdi_modifier
		}
		change_relations = {
		country = scope:target_country
		value = 10
		}

		save_scope_as = giver
		set_variable = giver

		trigger_event = { id = fdi.1 
		years = 1}
	}
	
	ai = {
		will_propose = { 
			always = yes
		}
	}
}


sponsor_terrorism = {
	requires_approval = no
	should_notify_third_parties = yes
	show_effect_in_tooltip = no

	potential = {
		NOT = { is_country_type = decentralized }
		scope:target_country = {
			NOT = { is_country_type = decentralized }
		}
		OR = {
			is_player = yes
			AND = {
			country_rank > rank_value:major_power 
			weekly_net_fixed_income > 10000

			OR = {
				has_attitude = {
					who = scope:target_country
					attitude = antagonistic
				}
				has_attitude = {
					who = scope:target_country
					attitude = belligerent
				}
				"relations(scope:target_country)" <= -30
				}
			}
		}
	}
	
	possible = {
		gold_reserves > 500000
		has_interest_marker_in_region = scope:target_country.capital.region
		NOT = { has_modifier = sponsoring_terrorism }
	}
	
	accept_effect = {
		add_treasury = -500000
		scope:target_country = {
			add_modifier = {
				name = sponsored_terrorism
				months = 36
			}
		}

			add_modifier = {
				name = sponsoring_terrorism
				months = 36
			}

		change_relations = {
			country = scope:target_country
			value = -10
		}
	}
	
	ai = {
		will_propose = { 
			always = yes
		}
	}
}

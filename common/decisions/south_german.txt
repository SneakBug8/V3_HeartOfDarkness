sgf_unite_aus = {
	is_shown = {
has_journal_entry = je_sgf_aus
	}
	possible = {
		is_at_war = no
		has_technology_researched = nationalism
		NOT = { has_modifier = sgf_test_modifier }
		custom_tooltip = {
			text = aus_sgf_relations_tt
			any_subject_or_below = {
				country_has_primary_culture = cu:south_german
				relations:root >= relations_threshold:amicable
				count >= 2
			}
		}
	}
	when_taken = {
		custom_tooltip = unite_sgf_tt
		hidden_effect = {
			trigger_event = { id = sgf_aus.2 popup = yes }
		}
		add_modifier = {
			name = sgf_test_modifier
			months = short_modifier_time
		}
	}

	ai_chance = {
		base = 0
		modifier = {
			has_technology_researched = pan-nationalism
			add = 100
		}
		modifier = {
			exists = c:NGF
			add = 100
		}		
	}
}

mexico_crown_emperor = {
	is_shown = {
		exists = c:MEX
		this = c:MEX
		NOT = { has_variable = mexico_crowned_emperor }
	}
	possible = {
		is_at_war = no
		is_subject = no

		exists = c:FRA
		c:FRA = {
				relations:root >= relations_threshold:amicable
			}
	}
	when_taken = {
		activate_law = law_type:law_monarchy
		activate_law = law_type:law_autocracy

        create_character = {
	            first_name = Maximilian
                last_name = von_Habsburg
                culture = cu:south_german
                female = no
                ruler = yes
                religion = catholic
                ideology = ideology_royalist
                interest_group = ig_landowners
                birth_date = 1832.7.06
                traits = {
                    ambitious demagogue
                }
			}

		c:FRA = {
			create_diplomatic_pact = {
                country = c:MEX
                type = protectorate
            }
		}
	}

	ai_chance = {
		base = 0
	}
}


portugal_return_brazil = {
	is_shown = {
		exists = c:POR
		this = c:POR

		exists = c:BRZ

		NOT = { has_variable = portugal_unification_war }
	}
	possible = {
		is_at_war = no
	}
	when_taken = {
		set_variable = portugal_unification_war
		create_diplomatic_play = {
                target_country = c:BRZ
				type = dp_annex_war
        }
		hidden_effect = { #Why? Because the Taiping Rebellion play grants an insane amount of infamy.
				change_infamy = -150
			}
	}

	ai_chance = {
		base = 0
	}
}


portugal_peaceful_reunification = {
	is_shown = {
		exists = c:POR
		this = c:POR

		exists = c:BRZ

		NOT = { has_variable = portugal_unification_war }
	}
	possible = {
		is_at_war = no

		c:BRZ.relations:root >= 60
	}
	when_taken = {
		set_variable = portugal_unification_war
		
		c:BRZ = {
			every_scope_state = {
				set_state_owner = c:POR
			}
		}
	}

	ai_chance = {
		base = 0
	}
}

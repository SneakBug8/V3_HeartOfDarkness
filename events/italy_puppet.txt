namespace = italy_puppet

# Italy absorbs other states
italy_puppet.1 = {
	type = country_event
	placement = ROOT

	title = can_aus.1.t
	desc = can_aus.1.d
	flavor = can_aus.1.f

	event_image = {
		video = "gfx/event_pictures/unspecific_signed_contract.bk2"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/unspecific/signed_contract"

	icon = "gfx/interface/icons/event_icons/event_map.dds"

	duration = 3

	trigger = {
        year > 1848

		OR = {
            AND = {
                exists = c:ITA
                c:ITA = {
                    is_at_war = no
                        has_technology_researched = nationalism
                    }
            }
            AND = {
                exists = c:SAR
                c:SAR = {
                    is_at_war = no

                    has_technology_researched = nationalism
                }
            }
            AND = {
                exists = c:SIC
                c:SIC = {
                    is_at_war = no

                    has_technology_researched = nationalism
                    }
            }
        }

        any_scope_state = {
            state_region = {
                    OR = {
                    is_homeland = cu:north_italian
                    is_homeland = cu:south_italian
                    }
                }
                NOT = {
                    owner = c:AUS
                }
            }
							
            NOT = { country_rank = rank_value:great_power }
        }

	option = {
		name = can_aus.1.a
		default_option = yes

        random_scope_state = {
            limit = {
                state_region = {
                    OR = {
                    is_homeland = cu:north_italian
                    is_homeland = cu:south_italian
                    }
                }
            }

            if = {
                limit = {
		            exists = c:ITA
                    c:ITA = {
                        has_technology_researched = nationalism
                    }
                }

                set_state_owner = c:ITA
            }
            else_if = {
                limit = {
                    exists = c:SAR
                    c:SAR = {
                        has_technology_researched = nationalism
                    }
                }

                set_state_owner = c:SAR
            }
            else_if = {
                limit = {
                    exists = c:SIC
                    c:SIC = {
                        has_technology_researched = nationalism
                    }
                }

                set_state_owner = c:SIC
            }
        }
	}
}

                owns_entire_state_region = STATE_KIRGHIZIA

        s:STATE_JETISY = {
            add_claim = c:RUS
        }
        s:STATE_TOMSK = {
            add_claim = c:RUS
        }
        s:STATE_KIRGHIZIA = {
            add_claim = c:RUS
        }

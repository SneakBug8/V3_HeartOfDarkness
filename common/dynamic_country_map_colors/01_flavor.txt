turkey = {
	color = "roman_red"

	possible = {
		AND = {
			exists = c:TUR
			THIS = c:TUR
		}
		OR = {
			has_law = law_type:law_parliamentary_republic
			has_law = law_type:law_presidential_republic
			has_law = law_type:law_council_republic
		}
	}
}


roman_rome = {
	color = "roman_red"

	possible = {
		exists = c:ITA
		THIS = c:ITA
		OR = {
			has_law = law_type:law_parliamentary_republic
			has_law = law_type:law_presidential_republic
			has_law = law_type:law_council_republic
		}
		was_formed_from = PAP
	}
}
